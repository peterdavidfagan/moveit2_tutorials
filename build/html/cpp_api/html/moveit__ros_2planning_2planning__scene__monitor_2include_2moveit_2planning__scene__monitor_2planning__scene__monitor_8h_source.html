689  void initialize(bool read_only)
690  {
691  if (planning_scene_monitor_)
692  lock_ = std::make_shared<SingleUnlock>(planning_scene_monitor_.get(), read_only);
693  }
694 
695  MOVEIT_STRUCT_FORWARD(SingleUnlock);
696 
697  // we use this struct so that lock/unlock are called only once
698  // even if the LockedPlanningScene instance is copied around
699  struct SingleUnlock
700  {
701  SingleUnlock(PlanningSceneMonitor* planning_scene_monitor, bool read_only)
702  : planning_scene_monitor_(planning_scene_monitor), read_only_(read_only)
703  {
704  if (read_only)
705  planning_scene_monitor_->lockSceneRead();
706  else
707  planning_scene_monitor_->lockSceneWrite();
708  }
709  ~SingleUnlock()
710  {
711  if (read_only_)
712  planning_scene_monitor_->unlockSceneRead();
713  else
714  planning_scene_monitor_->unlockSceneWrite();
715  }
716  PlanningSceneMonitor* planning_scene_monitor_;
717  bool read_only_;
718  };
719 
720  PlanningSceneMonitorPtr planning_scene_monitor_;
721  SingleUnlockPtr lock_;
722 };
