164 // ******************************************************************************************
165 // Highlight a robot group
166 // ******************************************************************************************
167 void RVizPanel::highlightGroupEvent(const std::string& group_name)
168 {
169  auto rm = getRobotModel();
170  if (!rm)
171  return;
172  // Highlight the selected planning group by looping through the links
173  if (!rm->hasJointModelGroup(group_name))
174  return;
175 
176  const moveit::core::JointModelGroup* joint_model_group = rm->getJointModelGroup(group_name);
177  if (joint_model_group)
178  {
179  const std::vector<const moveit::core::LinkModel*>& link_models = joint_model_group->getLinkModels();
180  // Iterate through the links
181  for (std::vector<const moveit::core::LinkModel*>::const_iterator link_it = link_models.begin();
182  link_it < link_models.end(); ++link_it)
183  highlightLink((*link_it)->getName(), QColor(255, 0, 0));
184  }
185 }

187 // ******************************************************************************************
188 // Unhighlight all robot links
189 // ******************************************************************************************
190 void RVizPanel::unhighlightAllEvent()
191 {
192  auto rm = getRobotModel();
193  if (!rm)
194  return;
195  // Get the names of the all links robot
196  const std::vector<std::string>& links = rm->getLinkModelNamesWithCollisionGeometry();
197 
198  // Quit if no links found
199  if (links.empty())
200  {
201  return;
202  }
203 
204  // check if rviz is ready
205  if (!rviz_manager_ || !robot_state_display_)
206  {
207  return;
208  }
209 
210  // Iterate through the links
211  for (std::vector<std::string>::const_iterator link_it = links.begin(); link_it < links.end(); ++link_it)
212  {
213  if ((*link_it).empty())
214  continue;
215 
216  robot_state_display_->unsetLinkColor(*link_it);
217  }
218 }
